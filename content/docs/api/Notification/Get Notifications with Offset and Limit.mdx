## Implementation Example

<div className="rounded-xl border border-indigo-500/30 bg-gradient-to-br from-indigo-500/10 to-indigo-500/5 p-6 my-6">
  <div className="flex items-center gap-3 mb-4">
    <span className="text-2xl">ðŸ”§</span>
    <div>
      <div className="font-bold text-lg">Infinite Scroll Implementation</div>
      <div className="text-sm text-muted-foreground">Example pagination logic</div>
    </div>
  </div>

  ```javascript
  // Initialize
  let currentOffset = 0;
  const pageSize = 20;
  let hasMore = true;

  // Load first page
  async function loadNotifications() {
    const response = await fetchGraphQL(`
      query {
        getNotifications(offset: ${currentOffset}, limit: ${pageSize}) {
          id
          title
          msg
          timeStamp
          read
        }
      }
    `);

    const notifications = response.data.getNotifications;

    // Check if there are more notifications
    if (notifications.length < pageSize) {
      hasMore = false;
    }

    // Update offset for next page
    currentOffset += notifications.length;

    return notifications;
  }

  // Load more when user scrolls
  async function loadMore() {
    if (hasMore) {
      const moreNotifications = await loadNotifications();
      appendToUI(moreNotifications);
    }
  }
  ```
</div>
